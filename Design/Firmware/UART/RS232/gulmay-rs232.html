char uart_rd;
//char uart_rd1;
int row=1,col=1;
// LCD module connections
sbit LCD_RS at LATD2_bit;
sbit LCD_EN at LATD3_bit;
sbit LCD_D4 at LATD4_bit;
sbit LCD_D5 at LATD5_bit;
sbit LCD_D6 at LATD6_bit;
sbit LCD_D7 at LATD7_bit;

sbit LCD_RS_Direction at TRISD2_bit;
sbit LCD_EN_Direction at TRISD3_bit;
sbit LCD_D4_Direction at TRISD4_bit;
sbit LCD_D5_Direction at TRISD5_bit;
sbit LCD_D6_Direction at TRISD6_bit;
sbit LCD_D7_Direction at TRISD7_bit;
// End LCD module connections

void main() {


  PPS_Mapping(100, _INPUT,  _U1RX);              // Sets pin RP100 to be Input, and maps U1RX to it
  PPS_Mapping(101, _OUTPUT, _U1TX);              // Sets pin RP101 to be Output, and maps U1TX to it
  
  PPS_Mapping(108, _INPUT,  _U2RX);              // Sets pin RP100 to be Input, and maps U1RX to it
  PPS_Mapping(109, _OUTPUT, _U2TX);              // Sets pin RP101 to be Output, and maps U1TX to it

  PLLFBD = 70;             // PLL multiplier M=70
  CLKDIV = 0x0000;         // PLL prescaler N1=2, PLL postscaler N2=2

  ANSELA = 0x00;
  ANSELB = 0x00;
  ANSELC = 0x00;
  ANSELD = 0x00;
  ANSELE = 0x00;
  ANSELG = 0x00;


  PORTD = 0x00 ;


    Lcd_Init(); //Initialize LCD
     Delay_ms(100);
     //Lcd_Cmd(_LCD_CLEAR); //Clear Display
     //Lcd_Cmd(_LCD_UNDERLINE_ON); //Underline fashion cursor
     Lcd_Out(1,1,"Gulmay Gen. Test");
     Delay_ms(1000);
     Lcd_Cmd(_LCD_CLEAR);

     UART1_Init(9600); //Initialize UART module
     Delay_ms(100);
     UART2_Init(9600); //Initialize UART module
     Delay_ms(100);
     /*UART1_Write_Text("Start");
     UART1_Write(13);
     UART1_Write(10); */
     
     UART1_Write_Text("?<M><CR>");



  while(1){
     if (UART2_Data_Ready()) {     // If data is received
      uart_rd = UART2_Read();     // read the received data
      UART2_Write(uart_rd);       // and send data via UART

    }
     
     if(UART1_Data_Ready()==1){ //If Data Ready,
     uart_rd=UART1_Read(); //Read it and store in variable i
     Lcd_Chr(row,col,uart_rd); //Print it in co-ordinates specified
     if(uart_rd==17){ //If ESC is pressed, clear display
     Lcd_Cmd(_LCD_CLEAR);
     col=1,row=1;
     }
     UART1_Write(uart_rd);
     col=col+1;
     }
     if(col==17 && row==1){ //On end of row, goto second row
     row=2;
     col=1;
     }
     if(row==2 && col==17){ //On end of display, clear display
     Lcd_Cmd(_LCD_CLEAR);
     col=1;
     row=1;
           }
     }
}
